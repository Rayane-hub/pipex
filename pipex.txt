
static void	ft_parent_way_cmd1(t_var *var, pid_t pid)
{
	int	status;

	printf("Pere : J'attends mon fils cmd1 qui a le PID recu de fork [%d].\n", pid);
	waitpid(pid, &status, 0);
	printf("Pere : Mon fils est sorti avec le statut %d\n\n", status);
	if (status != 0)
	{
		ft_close_fd(var);
		exit(EXIT_FAILURE);
	}
}
